Simplest smart contract ever

contract Board {
	string[] posts;
	int[] votes;
	address[] author;
	uint32 endblock = now + 30 days;

	function post(string text) {
		if (msg.value != 1 eth) return;
		posts.append(text);
		authors.append(msg.sender);
	}
	function vote(int index, int up) {
		if (abs(up) != 1) return;
		votes[index] += up;
	}
	function payout() {
		if (msg.blockNumber < endblock) return;
		int ix = max(votes);
		author[ix].send(this.balance);
	}
};
